public boolean validateData(String input)
{
      String goodPattern = "(\\w|\\d)+";
      int goodLength = 10;
      if(input.length() > goodLength)
      {
            return false;
      }
      Pattern p = Pattern.compile(goodPattern);
      Matcher m = p.matcher(input);
            if(!m.matches())
      {
            return false;
      }
      else
      {
            return true;
      }
}
