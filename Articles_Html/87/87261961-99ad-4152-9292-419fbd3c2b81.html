Solution Example

#! /usr/bin/env python3
import os
import sys
import stat
file = "temporary_file"
try:
    fd = os.open(file, os.O_CREAT | os.O_EXCL | os.O_WRONLY)
except OSError:
    print("File already exists.")
    sys.exit()

with os.fdopen(fd, 'w') as f:
    f.write('content')
